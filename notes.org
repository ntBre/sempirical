      SUBROUTINE PRINT_MATRIX(DESC, M, N, A, LDA)
      Character*(*) DESC
      Integer M, N, LDA
      Double Precision A(LDA, *)
      Integer I, J

      write(*,*)
      write(*,*) DESC
      do I = 1, M
